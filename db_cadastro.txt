CREATE TABLE PESSOAS (
    CHAVE_PESSOAS      NUMERIC(12,2) NOT NULL,
    CPF                VARCHAR(14),
    NOME               VARCHAR(120),
    DOCUMENTO          VARCHAR(80),
    ENDERECO           VARCHAR(200),
    BAIRRO             VARCHAR(50),
    CIDADE             VARCHAR(40),
    UF                 VARCHAR(2),
    TELEFONE           VARCHAR(40),
    TELEFONE1          VARCHAR(40),
    DATA_NASCIMENTO    TIMESTAMP,
    CHAVE_CONJUGE      NUMERIC(12,2),
    SEXO               VARCHAR(1),
    CEP                VARCHAR(12),
    EMAIL              VARCHAR(50),
    NOME_PAI           VARCHAR(120),
    NOME_MAE           VARCHAR(120),
    OBSERVACAO         VARCHAR(2000),
    NATURALIDADE       VARCHAR(80),
    PAIS               VARCHAR(60),
    ORGAO_EMISSOR      VARCHAR(20),
    UF_NASCIMENTO      VARCHAR(2),
    CIDADE_NASCIMENTO  VARCHAR(40),
    PAIS_RESIDENCIA    VARCHAR(100),
    NACIONALIDADE      VARCHAR(50)
);
CREATE TABLE PESSOAS_JURIDICAS (
    CHAVE_PESSOAS_JURIDICAS  NUMERIC(12,2) NOT NULL,
    SITUACAO                 VARCHAR(20),
    CNPJ                     VARCHAR(16),
    RAZAO_SOCIAL             VARCHAR(120),
    NOME_FANTASIA            VARCHAR(120),
    NATUREZA_JURIDICA        VARCHAR(100),
    CNAE                     VARCHAR(150),
    ENDERECO                 VARCHAR(200),
    BAIRRO                   VARCHAR(50),
    CIDADE                   VARCHAR(40),
    UF                       VARCHAR(2),
    CEP                      VARCHAR(12),
    EMAIL                    VARCHAR(50),
    TELEFONE                 VARCHAR(40),
    TELEFONE1                VARCHAR(40),
    DATA_ABERTURA            TIMESTAMP,
    OBSERVACAO               VARCHAR(2000),
    NUMERO_ALTER_CONTRATO    VARCHAR(30)
);
CREATE TABLE PESSOAS_VINCULOS (
    CHAVE_PESSOAS_VINCULOS   NUMERIC(12,2) NOT NULL,
    CHAVE_PESSOAS_JURIDICAS  NUMERIC(12,2),
    CHAVE_PESSOAS            NUMERIC(12,2),
    FUNCAO                   VARCHAR(50)
);
ALTER TABLE PESSOAS ADD PRIMARY KEY (CHAVE_PESSOAS);
ALTER TABLE PESSOAS_JURIDICAS ADD CONSTRAINT PESSOAS_JURIDICAS_PK PRIMARY KEY (CHAVE_PESSOAS_JURIDICAS);
ALTER TABLE PESSOAS_VINCULOS ADD CONSTRAINT PESSOAS_VINCULOS_PK PRIMARY KEY (CHAVE_PESSOAS_VINCULOS);
ALTER TABLE PESSOAS ADD CONSTRAINT PESSOAS_CONJUGE_FK FOREIGN KEY (CHAVE_CONJUGE) REFERENCES PESSOAS (CHAVE_PESSOAS);
ALTER TABLE PESSOAS_VINCULOS ADD CONSTRAINT PESSOAS_VINCULOS_JURIDICAS_FK FOREIGN KEY (CHAVE_PESSOAS_JURIDICAS) REFERENCES PESSOAS_JURIDICAS (CHAVE_PESSOAS_JURIDICAS);
ALTER TABLE PESSOAS_VINCULOS ADD CONSTRAINT PESSOAS_VINCULOS_PESSOAS_FK FOREIGN KEY (CHAVE_PESSOAS) REFERENCES PESSOAS (CHAVE_PESSOAS);
